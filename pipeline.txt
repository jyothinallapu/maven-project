pipeline{
    agent any
    tools{
        maven 'Localmaven'
    }
    stages{
        stage('Build'){
            steps {
                bat 'mvn clean package'
            }
            post {
                success {
                    echo 'Now Archiving...'
                    archiveArtifacts artifacts: '**/targets/*.war'
                }
            }
        }
    }
}
